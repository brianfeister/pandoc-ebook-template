  label/.style={
    inner sep=3pt,
    text width=\linewidth-2*3pt,
    node contents={\@hangfrom{#2}{#3}},
  },
]
\node (n) [label];
\begin{scope}
\clip (n.south west) rectangle (n.north east);
\node {};
% \node {\includegraphics{marble}};
\end{scope}
\draw [
  SectionUnderlineColor,
  draw,
  ultra thick,
] (n.south west) -- (n.south east);
\node [label];
\end{tikzpicture}%
}
\renewcommand{\sectionlinesformat}[4]{%
\ifstr{#1}{section}{% ---------------------------------------------------- H2 --
\begin{tikzpicture}[
  label/.style={
    inner sep=2pt,
    text width=\linewidth-2*2pt,
    node contents={#4},
  },
]
\node (n) [label];
\draw [
  draw,
  thick,
] (n.south west) -- (n.south east);
\end{tikzpicture}}{}%
\ifstr{#1}{subsection}{% ------------------------------------------------- H3 --
\begin{tikzpicture}[
  label/.style={
    fill,
    font=\color{white}\bfseries,
    inner sep=3pt,
    text width=\linewidth-2*3pt,
    node contents={#4},
  },
]
\node (n) [label];
\end{tikzpicture}}{}%
\ifstr{#1}{subsubsection}{% ---------------------------------------------- H4 --
\begin{tikzpicture}[
  label/.style={
    inner sep=2pt,
    text width=\linewidth-2*2pt,
    node contents={#4},
  },
]
\node (n) [label];
\draw [
  thick,
] (n.south west) -- (n.south east) (n.north west) -- (n.north east);
\end{tikzpicture}}{}%
